### 消息提示

共有四种样式 `success`、`info`、`warn`、`error`。

import { Playground, Props } from 'docz'
import { message } from '../components/message/index.tsx'
import Button from '../components/button/index.tsx'

<Playground>
  {() => {
    const onClick = (type, text) => {
      message[type](text)
    }
    return (
      <div>
        <Button type='success' onClick={() => onClick('success', '成功')}>
          success
        </Button>
        <Button type='danger' onClick={() => onClick('error', '失败')}>
          error
        </Button>
        <Button type='default' onClick={() => onClick('warn', '警告')}>
          warn
        </Button>
        <Button type='primary' onClick={() => onClick('info', '提示')}>
          info
        </Button>
      </div>
    )
  }}
</Playground>
