import { Playground, Props } from 'docz'
import { message } from '../components/message/index.tsx'

### 消息提示

共有四种样式 `success`、`info`、`warn`、`error`。

<Playground>
  {() => {
    const onClick = (type, text) => {
      message[type](text)
    }
    return (
      <div>
        <button onClick={() => onClick('success', '成功')}>success</button>
        <button onClick={() => onClick('error', '失败')}>error</button>
        <button onClick={() => onClick('warn', '警告')}>warn</button>
        <button onClick={() => onClick('info', '提示')}>info</button>
      </div>
    )
  }}
</Playground>
